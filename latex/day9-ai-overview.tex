\section{API Calls}
\begin{frame}{Steps}
\begin{itemize}
    \item Get an API key
\begin{itemize}
\item [OpenAI](https://platform.openai.com/)
\item [Gemini](https://ai.google.dev/gemini)
\item [Claude](https://docs.anthropic.com/en/api)
\item [Open Router](https://openrouter.ai/)
\end{itemize}
\item Write code that sends a prompt to an AI and gets a response.  
\item API key is included in code to authorize (for billing)
\item Write an interface that allows the user to formulate a prompt.  Your code may expand the prompt to include whatever information you want to provide to the AI.
\end{itemize}
\end{frame}
